Watch directory or file, the callback argument is a function with EventEmitter instance. See Watch Api Documentation

 
exports.watch = (options = {}, callback = ->) ->
  if not options.path
    options.path = process.cwd()
  else
    options.path = path.resolve process.cwd(), options.path
  watch.watch options.path, options, (event) ->
    event.on 'error', (error) ->
      callback error
    event.on 'change', (file) ->
      callback file
